3、编写输入三角形三边长a，b，c，求三角形的面积area程序
```
#include <stdio.h>   
#include <math.h>     
int main() 
{
	double a, b, c, s, area;
	printf("请输入三角形的三边长：\n");
	scanf("%lf  %lf  %lf", &a, &b, &c);
	if (a + b > c && a + c > b && b + c > a) {
		s = (a + b + c) / 2.0;
		area = sqrt(s * (s - a) * (s - b) * (s - c));
		printf("三角形面积为：%lf\n", area);
	}
	else {
		printf("这不是一个三角形。\n");
	}
	return 0;
}
```
